package com.pageobjects.parabank;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.support.FindBy;
import org.openqa.selenium.support.PageFactory;
import org.testng.Reporter;

public class RegisterForm 
{
    private WebDriver driver;

    @FindBy(linkText = "Register")
    private WebElement eleRegistor;

    @FindBy(id = "customer.firstName")
    private WebElement eleFirstName;

    @FindBy(id = "customer.lastName")
    private WebElement eleLastName;

    @FindBy(id="customer.address.street")
    private WebElement eleAddress;

    @FindBy(id="customer.address.city")
    private WebElement eleCity;

    @FindBy(id="customer.address.state")
    private WebElement eleState;

    @FindBy(id ="customer.address.zipCode")
    private WebElement eleZipcode;

    @FindBy(id = "customer.phoneNumber")
    private WebElement elePhone;

    @FindBy(id="customer.ssn")
    private WebElement elessn;

    @FindBy(name = "customer.username")
    private WebElement eleUsername;

    @FindBy(name = "customer.password")
    private WebElement elePassword;

    @FindBy(name = "repeatedPassword")
    private WebElement eleconfirm;

    @FindBy(className = "button")
    private WebElement eleRegisterButton;

    @FindBy(xpath = "//li[@class=\"contact\"]")
    private WebElement eleContact;


    public RegisterForm(WebDriver driver)
    {
        PageFactory.initElements(driver,this);
        this.driver=driver;
    }

    public void enterRegisterLink(){
        eleRegistor.click();
        Reporter.log("The Register link is clicked");
    }

    public void enterFirstname(String firstname){
        eleFirstName.sendKeys(firstname);
        Reporter.log("The First name is entered sucessfully" +firstname);
    }
    public void enterLastname(String lastname){
        eleLastName.sendKeys(lastname);
        Reporter.log("The Last name is entered sucessfully" +lastname);
    }
    public void enterAddress(String address){
        eleAddress.sendKeys(address);
        Reporter.log("The Address is entered sucessfully" +address);
    }
    public void enterCity(String city){
        eleCity.sendKeys(city);
        Reporter.log("The City name is entered sucessfully" +city);
    }
    public void enterState(String state){
        eleState.sendKeys(state);
        Reporter.log("The State is entered sucessfully" +state);
    }
    public void enterZipcode(String zipcode){
        eleZipcode.sendKeys(zipcode);
        Reporter.log("The zipcode is entered sucessfully" +zipcode);
    }
    public void enterPhone(String phone){
        elePhone.sendKeys(phone);
        Reporter.log("The Phone No.is entered sucessfully" +phone);
    }
    public void enterssn(String ssn){
        elessn.sendKeys(ssn);
        Reporter.log("The ssn is entered sucessfully" +ssn);
    }
    public void enterUsername(String username){
        eleUsername.sendKeys(username);
        Reporter.log("The Username is entered sucessfully" +username);
    }
    public void enterPassword(String password){
        elePassword.sendKeys(password);
        Reporter.log("The Password is entered sucessfully" +password);
    }
    public void enterConfirmPassword(String confirm){
        eleconfirm.sendKeys(confirm);
        Reporter.log("The confirm Password is entered sucessfully" +confirm);
    }
    public void clickRegisterButton() {
        eleRegisterButton.click();
        Reporter.log("The Register Button is clicked");
    }
   public void clickContactButton() {
        eleContact.click();
        Reporter.log("The contact Button is clicked");
    }
    }
