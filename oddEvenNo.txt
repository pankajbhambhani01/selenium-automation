public class EvenOddNo {
    public static void main(String[] args) {
        int[] ages = {1, 2, 3, 13, 4, 5};
        for(int i=0;i<ages.length; i++) {
            if (ages[i] % 2 == 0) {
                System.out.println("the no is even " + ages[i]);
            } else {
                System.out.println("the no is odd " + ages[i]);
            }
        }
    }
}
